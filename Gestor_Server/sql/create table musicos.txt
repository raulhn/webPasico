create table pasico.musico(
	nid integer primary key auto_increment,
	nombre varchar(300),
	primer_apellido varchar(300),
	segundo_apellido varchar(300),
	fecha_nacimiento date,
	
);

create table pasico.curso(
	nid integer primary key auto_increment,
	descripcion varchar(100) unique,
	año_inicio integer,
	año_fin integer
);

create table pasico.expediente(
	nid integer primary key auto_increment,
	nid_musico integer,
	nid_curso integer
);

alter table pasico.expediente add constraint fk_musico_expediente foreign key(nid_musico) references pasico.musico(nid);
alter table pasico.expediente add constraint fk_curso_expediente foreign key(nid_curso) references pasico.curso(nid);


create table pasico.instrumento(
	nid integer primary key auto_increment,
	descripcion varchar(150)
);

create table pasico.asignatura(
   nid integer primary key auto_increment,
   descripcion varchar(500),
   nid_instrumento integer
);

alter table pasico.asignatura add constraint fk_instrumento_asignatura foreign key(nid_instrumento) references pasico.instrumento(nid);

create table pasico.evaluacion_curso(
	nid integer primary key auto_increment,
	nid_expediente integer,
	nid_asignatura integer
);

alter table pasico.evaluacion_curso add constraint fk_expediente_evaluacion foreign key(nid_expediente) references pasico.expediente(nid);

create table pasico.calificaciones(
	nid integer primary key auto_increment,
	nid_evaluacion integer,
	trimestre integer,
	puntacion integer,
	observaciones varchar(1000)
);

alter table pasico.calificaciones add constraint fk_evaluacion_calificaciones foreign key(nid_evaluaicon) references pasico.evaluacion_curso(nid);
